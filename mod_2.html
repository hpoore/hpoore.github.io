In [76]:
total_vax = []
vax_completed = []
dates = []
kind = []
for i in range(0,len(countries)):
    total_vax.append(df["total_vaccinations"][df['country'] == countries[i]].iloc[-1,])
    vax_completed.append(df["people_fully_vaccinated"][df['country'] == countries[i]].iloc[-1,])  
    dates.append(df["date"][df['country'] == countries[i]].iloc[-1,])  
    kind.append(df["vaccines"][df['country'] == countries[i]].iloc[-1,])  
dataframe = pd.DataFrame({'total vax': total_vax, 'vax completed': vax_completed,"date":dates, 'vax' : kind}, index = countries)
